% Numeración automática de ejercicios
\newcounter{ejercicio}

\newenvironment{ejercicio}[1][]
{%
    \refstepcounter{ejercicio}%
    \begin{tcolorbox}[
        breakable,
        colback=white,
        colframe=black!55,
        title={Ejercicio \theejercicio\if\relax\detokenize{#1}\relax\else: #1\fi}
    ]
}
{%
    \end{tcolorbox}
}

}
