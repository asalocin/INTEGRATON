    \begin{equation*}
        \oint_{C} \frac{e^{iz}}{z^{2} + 1} \, dz
    \end{equation*}
    \begin{enumerate}
        \item Identificar los polos de la función integrando.
        \item Determinar cuáles de esos polos están dentro del contorno \(C\).
        \item Calcular los residuos en los polos dentro de \(C\).
        \item Aplicar el teorema de residuos para evaluar la integral.
    \end{enumerate}

\end{solucion}
